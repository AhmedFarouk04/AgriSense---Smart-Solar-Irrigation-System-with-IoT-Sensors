import { v } from "convex/values";
import { action } from "./_generated/server";

export const sendVerificationEmail = action({
  args: {
    email: v.string(),
    code: v.string(),
  },
  handler: async (ctx, args) => {
    const { email, code } = args;

    console.log(`Sending verification code ${code} to ${email}`);

    return { success: true };
  },
});
